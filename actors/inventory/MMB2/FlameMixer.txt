Actor FlameMixerWep : BaseMM8BDMWep
{
//$Category MM8BDME-Weapons
//$Title Flame Mixer
dropitem "FlameMixerWep"
Weapon.AmmoUse 1
Weapon.AmmoGive 27
Weapon.SlotNumber 4
Obituary "$OB_FLAMEMIXER"
Tag "$TAG_FLAMEMIXER"
Inventory.Pickupmessage "$PU_FLAMEMIXER"
weapon.ammotype "FlameMixerAmmo"
inventory.icon "FMIXICON"
States
{
SpawnLoop:
FMIX E -1
loop
Ready:
FMIX G 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_FLAMEMIXER)
FMIX GG 1 A_WeaponReady
Goto Ready+1
ReadyC:
Goto Ready+1
Deselect:
FMIX G 0
goto DeselectSwap
Select:
FMIX G 0
goto SelectSwap
Fire:
FMIX G 0 A_JumpIfNoAmmo("NoAmmo")
FMIX G 0 A_JumpIfInventory("IsSpecialBot", 1 ,"StoveManCheck")
FMIX A 0 A_JumpIfInventory("IsBot", 1, "BotCheck")
FMIX HI 2
FMIX G 0 A_PlaySoundEx("weapon/flamemixer","Weapon")
FMIX J 2 A_FireCustomMissile("FlameMixerSpawner_Start",0,1,0,0)
FMIX K 2 A_FireCustomMissile("FlameMixerSpawner",0,1,0,0)
FMIX L 2 A_FireCustomMissile("FlameMixerSpawner",0,1,0,0)
FMIX Z 0 A_SpawnItemEx("OilPitIgnite",0,0,8)
FMIX Z 20
FMIX E 0 A_Refire
Goto Ready+1
NoAmmo:
FMIX G 1 ACS_NamedExecuteAlways("core_noammo",0)
Goto Ready+1
StoveManCheck:
BVUL A 0 A_JumpIf(ACS_NamedExecuteWithResult("core_radiuspull", 0, 130, 0)>0, 2)
BVUL B 1 A_GiveInventory("FlameShowerWep",1)
BVUL B 1 A_SelectWeapon("FlameShowerWep")
Goto Fire+3
BotCheck:
SCLA A 0 A_JumpIf(ACS_NamedExecuteWithResult("Enc_StunCheck",1)==1, "BotFire")
FHUD A 3 //ACS_NamedExecuteAlways("Bots_WalkToAttack",0,1,1)
FHUD A 0 A_JumpIf(ACS_NamedExecuteWithResult("core_radiuspull", 0, 130, 0)>0, 2)
FHUD A 0 A_JumpIf(ACS_NamedExecuteWithResult("core_radiuspull", 0, 800, 0)>0, "BotCheck")
Goto Ready+1
Botfire:
FHUD A 0 ACS_NamedExecuteAlways("Enc_StateChange",0,2)
Goto Fire+3
}
}
