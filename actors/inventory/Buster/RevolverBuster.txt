
actor RevolvingUpgrade : BaseMM8BDMBusterUpgrade

}

actor RevolvingBuster : BaseMM8BDMWep 
{
Weapon.AmmoUse 5 
Weapon.AmmoGive 30 
Weapon.SlotNumber 1 
Obituary "$OB_REVOLVINGBUSTER" 
Tag "$TAG_REVOLVINGBUSTER"
Inventory.Pickupmessage "$PU_REVOLVINGUPGRADE"
weapon.ammotype "RevolvingBusterAmmo" 
inventory.pickupsound "weapon/weaponup" 
inventory.icon "NULLICON"
+WEAPON.AMMO_OPTIONAL
+WEAPON.DONTBOB
Scale 2.0
States
{
SpawnLoop: 
FAKE A 0 
Stop
Ready:
MAEB B 0 A_JumpIfInventory("BusterGiven",1,"TakeBusters")
Goto Ready2
TakeBusters:
TNT1 Q 0 A_GiveInventory("TakeBuster", 1)
Goto Ready2
Ready2:
FAKE B 0 ACS_NamedExecuteWithResult("core_weaponcolor",CLR_REVOLVINGBUSTER)
FAKE B 2 A_WeaponReady
Goto Ready2+1
Deselect: 
FAKE B 0
goto DeselectSwap
Select: 
FAKE B 0
Goto SelectSwap
Fire:
FAKE B 0 A_JumpIfNoAmmo("NoAmmo")
FAKE C 1 
FAKE B 0 A_PlaySoundEx("misc/fakeshot","Weapon") 
FAKE B 0 A_FireCustomMissile("RevolvingBusterProjectile",0,5,15,0) 
FAKE DE 1
FAKE B 0 A_Refire("FireH")
FAKE B 0 A_JumpIfNoAmmo("NoAmmo")
Goto Ready2+1
Altfire:
FAKE B 0 A_JumpIfInventory("ENC_ClassAltfire", 1, "ClassAltfire")
FAKE B 0 A_JumpIf(CallACS("enc_class_altfire"), "ClassAltfire")
FAKE B 1 A_JumpIfInventory("MM8BDMPlayer", 1, 1)
wait
FAKE B 0
FAKE B 0 A_JumpIfInventory("RevolvingBusterAmmo",30,"NoAmmo")
FAKE B 0 A_TakeInventory("RevolvingBusterAmmo",99)
FAKE B 0 A_PlaySoundEX("weapon/RevolvingBusterReload","Weapon")
FAKE CDEBCDEBCDEBCDE 2
FAKE BCDEBCDEBCDEBC 2 A_GiveInventory("RevolvingBusterAmmo",2)
FAKE B 0 A_GiveInventory("RevolvingBusterAmmo",30)
Goto Ready2+1
Noammo:
FAKE B 0 A_JumpIfInventory("RevolvingBusterAmmo",30,"FullAmmo")
FAKE B 0 A_TakeInventory("RevolvingBusterAmmo",99)
FAKE B 0 A_PlaySoundEX("weapon/RevolvingBusterReload","Weapon")
FAKE CDEBCDEBCDEBCDE 2
FAKE BCDEBCDEBCDEBC 2 A_GiveInventory("RevolvingBusterAmmo",2)
FAKE B 0 A_GiveInventory("RevolvingBusterAmmo",30)
Goto Ready2+1
FullAmmo:
FAKE B 0
Goto Ready2+1
FireH:
FAKE B 1 A_RailWait
FAKE B 0 A_JumpIfInventory("IsBot", 1, "BotFire")
FAKE B 0 A_Refire("FireH")
Goto Ready2+1
BotFire:
FAKE B 2 A_RailWait
Goto Ready2+1
ReadyC:
Goto Ready2+1
ClassAltfire:
FAKE B 0 
goto Super::Altfire
}
}

{
inventory.amount 1
inventory.maxamount 30
}
